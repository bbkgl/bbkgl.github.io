class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        unordered_map<int, int> dp;
        return dfs(amount, dp, coins);
    }

    int dfs(int n, unordered_map<int, int> &dp, vector<int> &coins) {
        if (n < 0) return -1;
        if (n == 0) return 0;
        int ans = INT_MAX;
        for (const int &it : coins) {
            if (dp[n - it] == 0) {
                int t = dfs(n - it, dp, coins);
                if (t >= 0) {
                    ans = min(dp[it] = t, ans);
                }
            }
        }
        if (ans != INT_MAX)
            return dp[n] = ans + 1;
        else
            return -1;
    }
};
