```cpp
#include <cstdio>

class A{
public:
    int a[2];

    A(int x, int y) : a{x, y} {}

    virtual void f() {};
};

int main() {
    A aa(22, 33);
    auto *a = (int *)&aa;
    printf("%d\n", a[2]);
    return 0;
}
```
